async handleAjaxRequest(options) {
    let attempt = 0;
    while (attempt < CONFIG.maxRetryAttempts) {
        try {
            return await this.performAjaxRequest(options);
        } catch (error) {
            // Intelligent retry with exponential backoff
            const delay = Math.min(
                CONFIG.retryDelayBase * Math.pow(2, attempt),
                CONFIG.maxRetryDelay
            );
            await this.delay(delay);
        }
    }
}
